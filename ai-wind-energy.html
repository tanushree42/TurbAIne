AI-Powered Wind Prediction Model


import numpy as np
def predict_wind_speed(data):
    return np.mean(data) * 1.2  # AI-powered optimization
wind_data = [10, 15, 14, 12, 16]
print("Predicted Wind Speed:", predict_wind_speed(wind_data))
